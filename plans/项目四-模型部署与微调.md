```python
# ============================================
# Cell 3: 加载模型（4bit 量化）
# ============================================
import torch
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    BitsAndBytesConfig,
    TrainingArguments,
)
from peft import LoraConfig, get_peft_model, TaskType

# 选模型：Qwen2.5-1.5B（T4 16GB 够用，4bit量化后约1.5GB显存）
model_name = "Qwen/Qwen2.5-1.5B-Instruct"

# 4bit 量化配置
bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_quant_type="nf4",           # NF4 量化类型
    bnb_4bit_compute_dtype=torch.float16,
    bnb_4bit_use_double_quant=True,      # 双重量化进一步节省显存
)

# 加载模型
model = AutoModelForCausalLM.from_pretrained(
    model_name,
    quantization_config=bnb_config,
    device_map="auto",
    trust_remote_code=True,
)
tokenizer = AutoTokenizer.from_pretrained(model_name, trust_remote_code=True)
tokenizer.pad_token = tokenizer.eos_token

print(f"模型加载完成！显存占用: {model.get_memory_footprint() / 1024**3:.1f} GB")
```

```python
# ============================================
# Cell 4: 配置 LoRA
# ============================================
# LoRA 核心思想：不修改原模型参数，只在特定层插入小的可训练矩阵
# 可训练参数通常 < 原模型的 1%

lora_config = LoraConfig(
    r=8,                    # LoRA 秩（rank），8 是常用值
    lora_alpha=16,          # 缩放参数，通常 = 2*r
    target_modules=["q_proj", "k_proj", "v_proj", "o_proj"],  # 对 attention 层加 LoRA
    lora_dropout=0.1,       # 防止过拟合
    bias="none",
    task_type=TaskType.CAUSAL_LM,
)

model = get_peft_model(model, lora_config)
model.print_trainable_parameters()
# 输出示例: trainable params: 2.3M || all params: 1.54B || trainable%: 0.15%
```

```python
# ============================================
# Cell 5: 数据预处理
# ============================================
def format_prompt(example):
    """把 instruction + input + output 拼接成训练文本"""
    return {
        "text": f"<|im_start|>system\n{example['instruction']}<|im_end|>\n"
                f"<|im_start|>user\n{example['input']}<|im_end|>\n"
                f"<|im_start|>assistant\n{example['output']}<|im_end|>"
    }

train_dataset = dataset["train"].map(format_prompt)
val_dataset = dataset["test"].map(format_prompt)

def tokenize(examples):
    result = tokenizer(
        examples["text"],
        truncation=True,
        padding="max_length",
        max_length=512,
    )
    result["labels"] = result["input_ids"].copy()
    return result

train_dataset = train_dataset.map(tokenize, batched=True)
val_dataset = val_dataset.map(tokenize, batched=True)

print(f"数据预处理完成！")
```

```python
# ============================================
# Cell 6: 开始训练
# ============================================
from transformers import Trainer

training_args = TrainingArguments(
    output_dir="./lora_output",
    num_train_epochs=5,               # 50条数据训练5轮，约15-30分钟
    per_device_train_batch_size=4,    # T4 16GB 可以用 batch=4
    per_device_eval_batch_size=4,
    gradient_accumulation_steps=2,    # 实际 batch = 4*2 = 8
    warmup_steps=10,
    learning_rate=2e-4,               # LoRA 常用学习率
    fp16=True,                        # T4 支持 fp16
    logging_steps=5,
    save_steps=50,
    eval_strategy="steps",
    eval_steps=25,
    save_total_limit=2,
    load_best_model_at_end=True,
    report_to="none",
)

trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=train_dataset,
    eval_dataset=val_dataset,
)

print("开始训练...")
trainer.train()
print("训练完成！")

# 保存 LoRA 权重
model.save_pretrained("./lora_manufacturing_final")
tokenizer.save_pretrained("./lora_manufacturing_final")
print("LoRA 权重已保存！")
```

```python
# ============================================
# Cell 7: 微调前后对比测试
# ============================================
test_questions = [
    "设备突然停机了怎么办？",
    "你们能开增值税专用发票吗？",
    "操作工违规操作导致事故，公司负责吗？",
    "设备噪音突然增大是什么原因？",
    "保修期内你们多久能上门维修？",
]

# 加载原始模型（不加载 LoRA）
base_model = AutoModelForCausalLM.from_pretrained(
    model_name,
    quantization_config=bnb_config,
    device_map="auto",
    trust_remote_code=True,
)

# 加载微调后的模型
from peft import PeftModel
finetuned_model = PeftModel.from_pretrained(base_model, "./lora_manufacturing_final")

print("=" * 60)
print("微调前后对比测试")
print("=" * 60)

for q in test_questions:
    print(f"\n📝 问题: {q}")
    
    # 微调前
    inputs = tokenizer(f"<|im_start|>user\n{q}<|im_end|>\n<|im_start|>assistant\n", return_tensors="pt").to("cuda")
    with torch.no_grad():
        base_output = base_model.generate(**inputs, max_new_tokens=150, temperature=0.3)
    base_answer = tokenizer.decode(base_output[0], skip_special_tokens=True)
    
    # 微调后
    with torch.no_grad():
        ft_output = finetuned_model.generate(**inputs, max_new_tokens=150, temperature=0.3)
    ft_answer = tokenizer.decode(ft_output[0], skip_special_tokens=True)
    
    print(f"\n🔴 微调前: {base_answer[-200:]}")
    print(f"\n🟢 微调后: {ft_answer[-200:]}")
    print("-" * 40)

# 预期效果：
# - 微调前：通用回答，没有制造业用语，可能拒绝专业问题
# - 微调后：使用制造业术语（工单、保修、停机、备件），回答更专业化
```

### 4.4 训练数据生成脚本（可选，0.5 天）

如果你不想手写 50 条数据，可以用 DeepSeek API 辅助生成：

**文件**：`lora_finetune/generate_data.py`

```python
"""用 DeepSeek API 批量生成制造业客服训练数据"""
from openai import OpenAI

client = OpenAI(api_key="...", base_url="https://api.deepseek.com")

SCENARIOS = [
    ("故障排查", "设备轴承温度过高、噪音异常、振动超标、漏油"),
    ("保修政策", "保修期限、保修范围、延保服务、退换货条件"),
    ("操作指导", "设备开机流程、参数设置、日常维护、关机步骤"),
    ("配件查询", "轴承型号、密封圈规格、滤芯更换周期、润滑油型号"),
    ("安全规范", "安全操作规程、防护装备要求、紧急停机流程、事故上报"),
    ("通用咨询", "公司联系方式、技术支持热线、产品资料索取"),
]

for scenario, topics in SCENARIOS:
    prompt = f"""请生成 8 条制造业客服对话训练数据。
场景：{scenario}
话题：{topics}

每条数据格式：
{{
    "instruction": "你是制造业客服助手。请根据用户问题给出专业回复。",
    "input": "用户的问题（口语化、真实）",
    "output": "客服的专业回复（包含制造业术语、操作步骤、安全提醒等）"
}}

要求：
- 回复要专业、具体、含有制造业术语
- 如果涉及操作，请列出编号步骤
- 如果涉及安全，请加入安全提醒
- 控制每条回复 100-200 字
"""
    response = client.chat.completions.create(
        model="deepseek-chat",
        messages=[{"role": "user", "content": prompt}],
        temperature=0.7,
    )
    print(response.choices[0].message.content)
    print("\n" + "="*60 + "\n")
```
